DECLARE

--    product details
    product_id NUMBER:=101;
    product_name VARCHAR2(20);

-- invertory details
    available_stock NUMBER:=5;
    requested_qut NUMBER:=10;

-- billing details
    product_price NUMBER:=2500;
    total_amount NUMBER;

-- customer details
    customer_name VARCHAR2(5);

-- discount
    discount_percent NUMBER:=0;
    final_amount NUMBER;

    chandu EXCEPTION;

BEGIN

    if 'available_stock' < 'requested_quantity' THEN
        RAISE chandu;.output
    end if;

    total_amount:=product_id*product_price;
    dbms_output.PUT_LINE('output: '|| total_amount);

EXCEPTION
    when chandu THEN
        dbms_output.put_line('requirement quantity is not available');
end;
/

declare

    a number :=10;
    b number :=5;
 
    total_stock NUMBER;
    custom EXCEPTION;
BEGIN  

    if a > b THEN
        raise custom;
    end if;

    total_stock:= a - b;
    dbms_output.put_line('output: '|| total_stock);

EXCEPTION
    when custom then
        dbms_output.put_line('required stock is not available');
end;
/
